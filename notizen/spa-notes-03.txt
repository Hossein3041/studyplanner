import { fetchSession } from "./api.js";

async function checkViewAuthorization(viewId) {
  const session = await fetchSession();
  const loggedIn = !!session.loggedIn;
  const role = session.role; // "STUDENT" oder "ADMIN"

  if (publicViews.includes(viewId)) {
    return viewId;
  }

  if (protectedViews.includes(viewId)) {
    if (!loggedIn) return "login";
    return viewId;
  }

  if (adminViews.includes(viewId)) {
    if (!loggedIn) return "login";
    if (role !== "ADMIN") return "dashboard"; // oder "home"
    return viewId;
  }

  // Fallback
  return "home";
}
